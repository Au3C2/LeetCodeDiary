class Solution:
    def findMaxAverage(self, nums: List[int], k: int) -> float:
        sumk = sum(nums[:k])
        ans = sumk
        for i in range(k,len(nums)):
            sumk = sumk - nums[i-k] + nums[i]
            ans = max(sumk,ans)
        return ans/k
    
# 数组，简单，每日一题。像是滑动窗口
