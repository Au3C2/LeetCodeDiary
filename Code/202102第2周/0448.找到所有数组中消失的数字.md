class Solution:
    def findDisappearedNumbers(self, nums: List[int]) -> List[int]:
        l = len(nums)
        for n in nums:
            nums[(n-1)%l] += l
        ans = list()
        for i,n in enumerate(nums):
            if n <= l:
                ans.append(i+1)
        return ans
