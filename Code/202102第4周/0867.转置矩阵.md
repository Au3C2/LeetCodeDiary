class Solution:
    def transpose(self, matrix: List[List[int]]) -> List[List[int]]:
        m, n = len(matrix), len(matrix[0])
        ans = [[0] *m for _ in range(n)]
        
        for i in range(0,m):
            for j in range(0,n):
                ans[j][i] = matrix[i][j]
        return ans
    
# 数组，简单，每日一题
