class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        n = len(prices)
        buy = prices[0]
        profit = 0
        for i in range(1, n):
            if prices[i] < buy:
                buy = prices[i]
            elif prices[i] > buy:
                profit += prices[i] - buy
                buy = prices[i]
        return profit

class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        dp0 = 0             # æ‰‹é‡Œæ²¡è‚¡ç¥¨
        dp1 = - prices[0]   # æ‰‹é‡Œæœ‰è‚¡ç¥¨
        for i in range(1, len(prices)):
            dp0 = max(dp0, dp1 + prices[i])
            dp1 = max(dp1, dp0 - prices[i])
        return dp0
