class Solution:
    def nextGreaterElements(self, nums: List[int]) -> List[int]:
        ans = list()
        l_nums = len(nums)
        for i,n in enumerate(nums):
            j = (i+1) % l_nums
            while nums[j]<=n:
                j = (j+1) % l_nums
                if j == i:
                    break
            if j == i:
                ans.append(-1)
            else:
                ans.append(nums[j])
        return ans
